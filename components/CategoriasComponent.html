import { Component, OnInit } from '@angular/core';
import { CategoriasService } from 'src/app/servicos/categorias.service';
import { Categorias } from 'src/app/modelos/Categorias';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-categorias',
  templateUrl: './categorias.component.html',
  styleUrls: ['./categorias.component.css']
})
export class CategoriasComponent implements OnInit {

  form: FormGroup

  modal: boolean = false

  categoria!: Categorias[]

  idCategoria!: any

  id: string | undefined
  tituloForm: string = 'Cadastrar Categoria'

  filter!:string

  constructor(private categoriaService:CategoriasService, private fb: FormBuilder) {
    this.form = this.fb.group({
      categoria: ['', [Validators.required,
                       Validators.minLength(3),
                       Validators.maxLength(60)
      ]]
    })
  }

  ngOnInit(): void {
    this.mostrarCategorias()

    this.categoriaService.getCategoriaEdit().subscribe(resultado => {
      this.id = resultado.id
      this.tituloForm = 'Editar Categoria'
      this.form.patchValue({
        categoria: resultado.categoria
      })
    })
  }

  cadastrarCategoria() {
    const categoria: Categorias = {
      categoria: this.form.value.categoria,
    }
    this.categoriaService.addCategoria(categoria).then(() => {
      console.log('Cadastrado!')
    }, error => {
      console.log(error)
    })

    console.log(this.form.value)
    this.form.reset()
  }

  mostrarCategorias() {
    this.categoriaService.listarCategoria().subscribe((doc) => {
      this.categoria = []
      doc.forEach((categoria:any) => {
        this.categoria.push({
          id: categoria.payload.doc.id,
          ...categoria.payload.doc.data()
        })
      })
    })
  }

  salvarCategoria() {
    if(this.id === undefined) {
      this.cadastrarCategoria()
    } else {
      this.editarCategoria(this.id)
    }
  }

  editarCategoria(id:string) {
    const categoria: any = {
      categoria: this.form.value.categoria
    }
    this.categoriaService.editarCategoria(id, categoria).then(() => {
      this.form.reset()
      this.tituloForm = 'Editar Categoria'
      this.id = undefined
      console.log('Categoria editada com sucesso!')
    }, error => {
      console.log(error)
    })
  }

  editar(categoria:Categorias) {
    this.categoriaService.mostrarCategoriaEdit(categoria)
  }

  cancelarAcao() {
    this.modal = false
  }

  confirmarAcao() {
    this.categoriaService.excluirCategoria(this.idCategoria).then(
      () => {
        console.log('Categoria excluida')
        this.modal = false
      },
      (error) => {
        console.log(error)
      }
    )
  }

  mostrarModal(id: any) {
    this.modal = true
    this.idCategoria = id
  }
}
