import { Component, OnInit } from '@angular/core';
import { Professor } from 'src/app/modelos/Professor';
import { ProfessorService } from './../../servicos/professor.service';
import { TurmaService } from './../../servicos/turma.service';
import { Turma } from './../../modelos/Turma';

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.css']
})
export class HomeComponent implements OnInit {

  professores: Professor[] = [];

  turmas!: Turma[]

  datas: any

  constructor(private professorService: ProfessorService, private turmaService:TurmaService) { }


  ngOnInit(): void {
    this.mostrarProfessor()
    this.mostrarTurmas()

  }

  statusTurma (){
    this.datas = this.turmas.map((item)=>{
      this.turmas = this.turmas || []
      const dataTurmas = this.turmas.map((element)=>{
        return {
          dataInicioSelecao: element.dataDeInicioProcesso,
          dataInicioCurso: element.dataDeInicioCurso,
          dataInicioProjetoFinal: element.dataDeInicioProjeto
        }
      })
      return dataTurmas
      console.log(dataTurmas);
    })

  }

  mostrarProfessor() {
    this.professorService.listarProfessor().subscribe((doc) => {
      this.professores = [];
      doc.forEach((element: any) => {
          this.professores.push({
            id: element.payload.doc.id,
            ...element.payload.doc.data(),
          });
      });
    });
  }

  mostrarTurmas() {
    this.turmaService.listarTurma().subscribe((doc) => {

      this.turmas = []
      doc.forEach((turmas:any) => {
        // console.log(turmas.payload.doc.professor1.id);

        this.turmas.push({
          id: turmas.payload.doc.id,
          ...turmas.payload.doc.data()
        })
      })
    })
  }


}
