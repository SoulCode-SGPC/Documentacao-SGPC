import { ProfessorService } from 'src/app/servicos/professor.service';
import { Professor } from 'src/app/modelos/Professor';
import { Router } from '@angular/router';
import { Component, OnInit } from '@angular/core';
import { TurmaService } from './../../servicos/turma.service';
import { Turma, Professores } from './../../modelos/Turma';

@Component({
  selector: 'app-lista-turma',
  templateUrl: './lista-turma.component.html',
  styleUrls: ['./lista-turma.component.css'],
})
export class ListaTurmaComponent implements OnInit {
  modal: boolean = false;

  idTurma!: any;

  turmas!: Turma[];

  filter!: string;

  professores!: Professor[];

  constructor(
    private turmaService: TurmaService,
    private router: Router,
    private professorService: ProfessorService
  ) {}

  ngOnInit(): void {
    this.mostrarTurmas();
  }

  mostrarTurmas() {
    this.turmaService.listarTurma().subscribe((doc) => {
      this.turmas = [];
      doc.forEach((turmas: any) => {
        // console.log(turmas.payload.doc.professor1.id);

        this.turmas.push({
          id: turmas.payload.doc.id,
          ...turmas.payload.doc.data(),
        });
      });
      this.profComTurma();
    });
  }

  mostrarProfessor() {
    this.professorService.listarProfessor().subscribe((doc) => {
      console.log(doc);
      this.professores = [];
      doc.forEach((element: any) => {
        this.professores.push({
          id: element.payload.doc.id,
          ...element.payload.doc.data(),
        });
      });
    });
  }


  profComTurma() {

    let hoje = new Date().getTime();
    this.turmas.forEach((e) => {
      let inicioProjeto = new Date(e.dataDeInicioProjeto).getTime() + 10800000;
      let finalCurso = new Date(e.dataDeTerminoCurso).getTime() + 10800000;
      console.log(hoje);
      console.log(inicioProjeto);
      console.log(finalCurso);

      if (hoje >= inicioProjeto && hoje < finalCurso) {
        let professorAtual: Professor = {
          id: e.professor1.id,
          disponibilidade: 50,
        };
        this.professorService
          .AttDisp(<any>professorAtual.id, professorAtual)
          .then(
            () => {
              console.log('Professor atualizado disponibilidade');
            },
            (error) => {
              console.log(error);
            }
          );
        let professorAtual2: Professor = {
          id: e.professor2.id,
          disponibilidade: 50,
        };
        this.professorService
          .AttDisp(<any>professorAtual2.id, professorAtual2)
          .then(
            () => {
              console.log('Professor atualizado disponibilidade');
            },
            (error) => {
              console.log(error);
            }
          );
      } else if (hoje > finalCurso) {
        let professorDisponivel: Professor = {
          id: e.professor1.id,
          disponibilidade: 100,
        };
        this.professorService.AttDisp(
          <any>professorDisponivel.id,
          professorDisponivel
        );
        let professorDisponivel2: Professor = {
          id: e.professor2.id,
          disponibilidade: 100,
        };
        this.professorService.AttDisp(
          <any>professorDisponivel2.id,
          professorDisponivel2
        );
      }
    });

  }

  cancelarExclusao() {
    this.modal = false;
  }

  confirmarExclusao() {
    this.turmas.forEach((e) => {
      let dispParaTurmaEStatus: any = {
        id: this.idTurma,
        professor1: {
          id: e.professor1.id,
          nome: e.professor1.nome,
          disponibilidade: 0,
        },
        professor2: {
          id: e.professor2.id,
          nome: e.professor2.nome,
          disponibilidade: 0,
        },
        status: false,
      };
      this.turmaService
        .editarTurma(dispParaTurmaEStatus.id, dispParaTurmaEStatus)
        .then(
          () => {
            console.log('Turma desativada e profs indisponíveis dentro dela!');
            this.modal = false;
          },
          (error) => {
            console.log(error);
          }
        );

      let dispParaFora1: any = {
        id: e.professor1.id,
        disponibilidade: 100,
      };
      this.professorService.AttDisp(dispParaFora1.id, dispParaFora1).then(
        () => {
          console.log('Professor1 disponível para nova turma!');
          this.modal = false;
        },
        (error) => {
          console.log(error);
        }
      );
      let dispParaFora2: any = {
        id: e.professor2.id,
        disponibilidade: 100,
      };
      this.professorService.AttDisp(dispParaFora2.id, dispParaFora2).then(
        () => {
          console.log('Professor2 disponível para nova turma!');
          this.modal = false;
        },
        (error) => {
          console.log(error);
        }
      );
    });
  }

  mostrarModal(id: any) {
    this.modal = true;
    this.idTurma = id;
  }

  // editarTurma(turma:Turma) {
  //   this.turmaService.mostrarTurmaEdit(turma)
  //   this.router.navigate(['turma/cadastro'])
  // }

  editarTurma(id: any) {
    this.turmaService.mostrarTurmaEdit(id);
    this.router.navigate(['/turma/cadastro/' + id]);
  }
}
