import { Professor } from 'src/app/modelos/Professor';
import { ProfessorService } from './../../servicos/professor.service';
import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { TecnologiasService } from 'src/app/servicos/tecnologias.service';
import { Tecnologias } from 'src/app/modelos/Tecnologias';

@Component({
  selector: 'app-form-professor',
  templateUrl: './form-professor.component.html',
  styleUrls: ['./form-professor.component.css'],
})
export class FormProfessorComponent implements OnInit {

  form: FormGroup;
  id: string | undefined;
  tituloForm: string = 'Cadastrar Professor';
  professor!: Professor[];
  tecnologias!: Tecnologias[];
  techCheckBox:any = []


  constructor(private service: ProfessorService, private fb: FormBuilder,private route:Router, private activedroute: ActivatedRoute, private tecnologiaService: TecnologiasService) {
    this.form = this.fb.group({
      nome: [
        '',
        [
          Validators.required,
          Validators.minLength(3),
          Validators.maxLength(100),
          Validators.pattern(/^(?!À-Ö)[A-Za-z\',\-àáâãçèéêẽíôõóúû ]*$/),
        ],
      ],
      formacao: [
        '',
        [
          Validators.required,
          Validators.pattern(/^(?!À-Ö)[A-Za-z\',\-àáâãçèéêẽíôõóúû ]*$/),
        ],
      ],
      email: ['', [Validators.email, Validators.required]],
      contato: ['', [Validators.required]],
      tecnologia: ['', [this.ValidateCheckBox]],
    });
  }

  ValidateCheckBox(control: any): { [key: string]: any } | null | boolean {
    control = document.getElementsByName('tecnologia');
    console.log(control);
    let validei: boolean = false
    for (let i = 0; i < control.length; i++) {
      if (control[i].checked) {
        console.log(control[i].checked);
        i = control.length;
       return validei = true ;
      }
    }
    return {'tecnologia':validei}
  }

  ngOnInit(): void {
    this.mostrarTecnologias()
   }

  cadastrarProfessor() {
    const professor: Professor = {
      nome: this.form.value.nome,
      formacao: this.form.value.formacao,
      email: this.form.value.email,
      contato: this.form.value.contato,
      status: true,
      tecnologia: this.techCheckBox,
      disponibilidade: 100
    };
    console.log(this.techCheckBox);

    this.service.addProfessor(professor).then(
      () => {
        console.log('Sucesso!');
        this.route.navigate(['/professor/lista'])
      },
      (error) => {
        console.log(error);
      }
    );
    console.log(professor);
  }

  mostrarTecnologias() {
    this.tecnologiaService.listarTecnologia().subscribe((doc) => {
      this.tecnologias = []
      doc.forEach((tecnologia:any) => {
        this.tecnologias.push({
          id: tecnologia.payload.doc.id,
          ...tecnologia.payload.doc.data()
        })
      })
    })
  }

  qualquerNome(event: any) {
    if (this.techCheckBox.includes(event.target.value) == false) {
      this.techCheckBox.push(event.target.value);
      console.log(this.techCheckBox);
    } else {
      this.techCheckBox.splice(
        this.techCheckBox.indexOf(event.target.value),
        1
      );
      console.log(this.techCheckBox);
    }
  }

}
