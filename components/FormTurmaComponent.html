import { Turma } from './../../modelos/Turma';
import { TurmaService } from './../../servicos/turma.service';
import { Component, OnInit, NgModule } from '@angular/core';
import {
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { Router } from '@angular/router';
import { Tecnologias } from 'src/app/modelos/Tecnologias';
import { TecnologiasService } from 'src/app/servicos/tecnologias.service';
import { Professor } from 'src/app/modelos/Professor';
import { ProfessorService } from 'src/app/servicos/professor.service';
import { Categorias } from 'src/app/modelos/Categorias';
import { CategoriasService } from 'src/app/servicos/categorias.service';


@Component({
  selector: 'app-form-turma',
  templateUrl: './form-turma.component.html',
  styleUrls: ['./form-turma.component.css'],
})
export class FormTurmaComponent implements OnInit {
  [x: string]: any;
  form: FormGroup;
  id: string | undefined;
  tituloForm: string = 'Cadastrar Turma';
  tecnologias!: Tecnologias[];
  professores!: Professor[];
  turmas!: Turma[];
  categoria!: Categorias[];
  idProcurado!: any;
  nomeProfessor1 = '';
  nomeProfessor2 = '';
  nomeEIdUm!: any;
  separandoNomeEIdUm!: any;
  nomeEIdDois!: any;
  separandoNomeEIdDois!: any;
  dataFormataString!: any

  techCheckBox: any = [];

  turmasExistente: any = []

  constructor(
    private service: TurmaService,
    private fb: FormBuilder,
    private route: Router,
    private tecnologiaService: TecnologiasService,
    private professorService: ProfessorService,
    private categoriaService: CategoriasService,
  ) {
    this.form = this.fb.group({
      nome: [
        '',
        [
          Validators.required,
          Validators.minLength(3),
          Validators.maxLength(100),
          Validators.pattern(/^(?!À-Ö)[A-Za-z0-9\',\-àáâãçèéêẽíôõóúû ]*$/),
        ],
      ],
      categoria: ['', [Validators.required]],
      professor1: ['', [Validators.required]],
      idProfessorUm: [''],
      disponibilidadeProfessor1: [''],
      professor2: [''],
      nomeProfessor2: [''],
      disponibilidadeProfessor2: [''],
      dataDeInicioProcesso: [''],
      dataDeTerminoProcesso: [''],
      dataDeInicioCurso: [''],
      dataDeTerminoCurso: [''],
      dataDeInicioProjeto: [''],
      dataDeTerminoProjeto: [''],
      tecnologia: ['', [this.ValidateCheckBox]],
    });
  }

  ValidateCheckBox(control: any): { [key: string]: any } | null | boolean {
    control = document.getElementsByName('tecnologia');

    let validei: boolean = false;
    for (let i = 0; i < control.length; i++) {
      if (control[i].checked) {
        console.log(control[i].checked);
        i = control.length;
        return (validei = true);
      }
    }
    return { tecnologia: validei };
  }

  ngOnInit(): void {
    this.mostrarTecnologias();
    this.mostrarProfessor();
    this.mostrarCategorias();
  }

  cadastrarTurma() {
    this.nomeEIdUm = this.form.value.professor1;
    this.separandoNomeEIdUm = this.nomeEIdUm.split('-');
    this.nomeEIdDois = this.form.value.professor2;
    this.separandoNomeEIdDois = this.nomeEIdDois.split('-');

    this.service.listarTurma().subscribe((doc) => {
      this.turmas = []

      doc.forEach((turmas: any) => {
        this.turmas.push({
          id: turmas.payload.doc.id,
          ...turmas.payload.doc.data()
        })
      })

      this.turmas.forEach((e: any) => {
        this.turmasExistente.push(e.nome)
      })

      console.log(this.form.value.dataDeTerminoCurso);
      console.log(this.form.value.dataDeTerminoProjeto);

      const turma: Turma = {
        nome: this.form.value.nome,
        categoria: this.form.value.categoria,
        professor1: {
          id: this.separandoNomeEIdUm[0],
          nome: this.separandoNomeEIdUm[1],
          disponibilidade: 100,
        },
        professor2: {
          id: this.separandoNomeEIdDois[0],
          nome: this.separandoNomeEIdDois[1],
          disponibilidade: 100,
        },
        tecnologias: this.techCheckBox,
        dataDeInicioProcesso: this.form.value.dataDeInicioProcesso,
        dataDeInicioCurso: this.form.value.dataDeInicioCurso,
        dataDeInicioProjeto: this.form.value.dataDeInicioProjeto,
        dataDeTerminoProcesso: this.form.value.dataDeTerminoProcesso,
        dataDeTerminoCurso: this.form.value.dataDeTerminoCurso,
        dataDeTerminoProjeto: this.form.value.dataDeTerminoProjeto,
        status: true
      };

      console.log(turma.dataDeInicioCurso);
      console.log(turma.dataDeTerminoCurso);
      console.log(turma.dataDeTerminoProjeto);

      if (turma.professor1 != turma.professor2 && this.turmasExistente.includes(this.form.value.nome) == false) {
        this.service.addTurma(turma).then(
          () => {
            const disponibilidade = { disponibilidade: 0 };
            this.professorService.AttDisp(
              this.separandoNomeEIdUm[0],
              disponibilidade
            );
            this.professorService.AttDisp(
              this.separandoNomeEIdDois[0],
              disponibilidade
            );
            console.log('Sucesso!');

            this.route.navigate(['/turma/lista']);
          },
          (error) => {
            console.log(error);
          }
        );
      } else {
        console.log('professor ja existe ou mesmo professor esta sendo cadastrado')
      }
    })
  }

  mostrarTecnologias() {
    this.tecnologiaService.listarTecnologia().subscribe((doc) => {
      this.tecnologias = [];
      doc.forEach((tecnologia: any) => {
        this.tecnologias.push({
          id: tecnologia.payload.doc.id,
          ...tecnologia.payload.doc.data(),
        });
      });
    });
  }

  mostrarProfessor() {
    this.professorService.listarProfessor().subscribe((doc) => {
      this.professores = [];
      doc.forEach((element: any) => {
        this.professores.push({
          id: element.payload.doc.id,
          ...element.payload.doc.data(),
        });
      });
    });
  }

  mostrarCategorias() {
    this.categoriaService.listarCategoria().subscribe((doc) => {
      this.categoria = [];
      doc.forEach((categoria: any) => {
        this.categoria.push({
          id: categoria.payload.doc.id,
          ...categoria.payload.doc.data(),
        });
      });
    });
  }

  qualquerNome(event: any) {
    if (this.techCheckBox.includes(event.target.value) == false) {
      this.techCheckBox.push(event.target.value);
      console.log(this.techCheckBox);
    } else {
      this.techCheckBox.splice(
        this.techCheckBox.indexOf(event.target.value),
        1
      );
      console.log(this.techCheckBox);
    }
  }
}
