import { CategoriasService } from './../../servicos/categorias.service';
import { Categorias } from './../../modelos/Categorias';
import { ProfessorService } from './../../servicos/professor.service';
import { TecnologiasService } from './../../servicos/tecnologias.service';
import { Professor } from './../../modelos/Professor';
import { Tecnologias } from './../../modelos/Tecnologias';
import { ActivatedRoute, Router } from '@angular/router';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { TurmaService } from './../../servicos/turma.service';
import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-edit-turma',
  templateUrl: './edit-turma.component.html',
  styleUrls: ['./edit-turma.component.css'],
})
export class EditTurmaComponent implements OnInit {
  form: FormGroup;
  id!: string;
  tecnologias!: Tecnologias[];
  professores!: Professor[];
  categoria!: Categorias[];
  turma: any = [];
  professor1: any = [];
  professor2: any = [];
  nomeEIdUm: any;
  separandoNomeEIdUm: any;
  nomeEIdDois: any;
  turmaControle: any = [];
  checked: boolean = true;
  techCheckBox: any = [];

  constructor(
    private service: TurmaService,
    private fb: FormBuilder,
    private activatedRoute: ActivatedRoute,
    private route: Router,
    private tecnologiaService: TecnologiasService,
    private professorService: ProfessorService,
    private categoriaService: CategoriasService
  ) {
    this.form = this.fb.group({
      nome: [
        '',
        [
          Validators.required,
          Validators.minLength(3),
          Validators.maxLength(100),
          Validators.pattern(/^(?!À-Ö)[A-Za-z0-9\',\-àáâãçèéêẽíôõóúû ]*$/),
        ],
      ],
      categoria: ['', [Validators.required]],
      tecnologia: [],
      dataDeInicioProcesso: ['', [Validators.required]],
      dataDeTerminoProcesso: ['', [Validators.required]],
      dataDeInicioCurso: ['', [Validators.required]],
      dataDeTerminoCurso: ['', [Validators.required]],
      dataDeInicioProjeto: ['', [Validators.required]],
      dataDeTerminoProjeto: ['', [Validators.required]],
    });
  }

  marcados: any = [];

  ngOnInit(): void {
    this.mostrarTecnologias();
    const id_entrada = <any>this.activatedRoute.snapshot.params['id'];
    this.service.listarUmaTurma(id_entrada).subscribe((res) => {
      console.log(res.data(), res.id);
      let resultado = { id: res.id, ...res.data() };
      console.log(resultado);
      this.id = resultado.id;

      this.turma = resultado;
      this.turmaControle = resultado;
      this.professor1 = resultado.professor1;
      this.professor2 = resultado.professor2;
      console.log(this.professor2);

      this.marcados = resultado.tecnologias
      console.log(this.marcados);

      this.form.patchValue({
        nome: resultado.nome,
        categoria: resultado.categoria,
        dataDeInicioProcesso: resultado.dataDeInicioProcesso,
        dataDeInicioCurso: resultado.dataDeInicioCurso,
        dataDeInicioProjeto: resultado.dataDeInicioProjeto,
        dataDeTerminoProcesso: resultado.dataDeTerminoProcesso,
        dataDeTerminoCurso: resultado.dataDeTerminoCurso,
        dataDeTerminoProjeto: resultado.dataDeTerminoProjeto,
      });
    });
    this.mostrarProfessor();
    this.mostrarCategorias();
  }

  editarTurma(id: string) {
    this.marcados.forEach((e: any) => {
      if (this.techCheckBox.includes(e) == false) {
        this.techCheckBox.push(e);
      }
    });
    const turma: any = {
      nome: this.form.value.nome,
      categoria: this.form.value.categoria,
      professor1: this.professor1,
      professor2: this.professor2,
      tecnologias: this.techCheckBox,
      dataDeInicioProcesso: this.form.value.dataDeInicioProcesso,
      dataDeInicioCurso: this.form.value.dataDeInicioCurso,
      dataDeInicioProjeto: this.form.value.dataDeInicioProjeto,
      dataDeTerminoProcesso: this.form.value.dataDeTerminoProcesso,
      dataDeTerminoCurso: this.form.value.dataDeTerminoCurso,
      dataDeTerminoProjeto: this.form.value.dataDeTerminoProjeto,
    };
    this.turma = turma;
    this.turma.professor1 = this.professor1;
    this.turma.professor2 = this.professor2;

    this.service.editarTurma(id, this.turma).then(
      () => {
        if (this.turma.professor1.id != this.turmaControle.professor1.id) {
          let professorDisponivel: Professor = {
            id: this.turma.professor1.id,
            disponibilidade: 0,
          };
          console.log(professorDisponivel);

          this.professorService.AttDisp(
            <any>professorDisponivel.id,
            professorDisponivel
          );
          let professorLiberado: Professor = {
            id: this.turmaControle.professor1.id,
            disponibilidade: 100,
          };
          console.log(professorLiberado);

          this.professorService.AttDisp(
            <any>professorLiberado.id,
            professorLiberado
          );
        }

        if (this.turma.professor2.id != this.turmaControle.professor2.id) {
          let professorLiberado2: Professor = {
            id: this.turmaControle.professor2.id,
            disponibilidade: 100,
          };
          console.log(professorLiberado2);

          this.professorService.AttDisp(
            <any>professorLiberado2.id,
            professorLiberado2
          );
          let professorDisponivel2: Professor = {
            id: this.turma.professor2.id,
            disponibilidade: 0,
          };
          this.professorService.AttDisp(
            <any>professorDisponivel2.id,
            professorDisponivel2
          );
        }

        console.log('Turma editada com sucesso');
        this.route.navigate(['/turma/lista']);

      },
      (error) => {
        console.log(error);
      }
    );

    console.log(this.turma);
  }

  qualquerNome(event: any) {
    if (this.marcados.includes(event.target.value) == false) {
      if (this.techCheckBox.includes(event.target.value) == false) {
        this.techCheckBox.push(event.target.value);
        console.log(this.techCheckBox);
      } else {
        this.techCheckBox.splice(
          this.techCheckBox.indexOf(event.target.value),
          1
        );
        console.log(this.techCheckBox);
      }
    } else {
      this.marcados.splice(this.marcados.indexOf(event.target.value), 1);
    }
  }

  mostrarTecnologias() {
    this.tecnologiaService.listarTecnologia().subscribe((doc) => {
      this.tecnologias = [];
      doc.forEach((tecnologia: any) => {
        this.tecnologias.push({
          id: tecnologia.payload.doc.id,
          ...tecnologia.payload.doc.data(),
        });
      });
    });
  }

  mostrarProfessor() {
    this.professorService.listarProfessor().subscribe((doc) => {
      this.professores = [];
      doc.forEach((element: any) => {
        this.professores.push({
          id: element.payload.doc.id,
          ...element.payload.doc.data(),
        });
      });
    });
  }

  mostrarCategorias() {
    this.categoriaService.listarCategoria().subscribe((doc) => {
      this.categoria = [];
      doc.forEach((categoria: any) => {
        this.categoria.push({
          id: categoria.payload.doc.id,
          ...categoria.payload.doc.data(),
        });
      });
    });
  }

  trocaProfessor1(event: any) {
    this.nomeEIdUm = event.target.value;
    this.separandoNomeEIdUm = this.nomeEIdUm.split('-');
    this.nomeEIdDois = event.target.value;
    this.separandoNomeEIdUm = this.nomeEIdDois.split('-');

    this.professor1 = {
      id: this.separandoNomeEIdUm[0],
      nome: this.separandoNomeEIdUm[1],
      disponibilidade: 100,
    };
  }

  trocaProfessor2(event: any) {
    this.nomeEIdUm = event.target.value;
    this.separandoNomeEIdUm = this.nomeEIdUm.split('-');

    this.professor2 = {
      id: this.separandoNomeEIdUm[0],
      nome: this.separandoNomeEIdUm[1],
      disponibilidade: 100,
    };
  }
}
