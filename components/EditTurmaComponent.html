import { CategoriasService } from './../../servicos/categorias.service';
import { Categorias } from './../../modelos/Categorias';
import { ProfessorService } from './../../servicos/professor.service';
import { TecnologiasService } from './../../servicos/tecnologias.service';
import { Professor } from './../../modelos/Professor';
import { Tecnologias } from './../../modelos/Tecnologias';
import { ActivatedRoute, Router } from '@angular/router';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { TurmaService } from './../../servicos/turma.service';
import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-edit-turma',
  templateUrl: './edit-turma.component.html',
  styleUrls: ['./edit-turma.component.css']
})
export class EditTurmaComponent implements OnInit {
  form: FormGroup;
  id!: string;
  tecnologias!: Tecnologias[]
  professores!: Professor[]
  categoria!: Categorias[]

  constructor(private service: TurmaService, private fb: FormBuilder, private activatedRoute : ActivatedRoute, private route: Router, private tecnologiaService: TecnologiasService, private professorService: ProfessorService, private categoriaService: CategoriasService) {
    this.form = this.fb.group({
      nome: ['', [Validators.required,
                  Validators.minLength(3),
                  Validators.maxLength(100),
                  Validators.pattern(/^(?!À-Ö)[A-Za-z0-9\',\-àáâãçèéêẽíôõóúû ]*$/)]],
      categoria: ['', [Validators.required]],
      professor1: ['', [Validators.required]],
      professor2: ['', [Validators.required]],
      tecnologia: ['', [Validators.required]],
      dataDeInicioProcesso: ['',[Validators.required]],
      dataDeTerminoProcesso: ['',[Validators.required]],
      dataDeInicioCurso: ['',[Validators.required]],
      dataDeTerminoCurso: ['',[Validators.required]],
      dataDeInicioProjeto: ['',[Validators.required]],
      dataDeTerminoProjeto: ['',[Validators.required]],
    });
   }

  ngOnInit(): void {
    const id_entrada = <any>this.activatedRoute.snapshot.params['id']
    this.service.listarUmaTurma(id_entrada).subscribe((res) =>{
      console.log(res.data(), res.id)
      let resultado = {id:res.id,...res.data()}
      console.log(resultado)
      this.id = resultado.id

      


      this.form.patchValue({
        nome: resultado.nome,
        categoria: resultado.categoria,
        professor1: resultado.professor1,
        professor2: resultado.professor2,
        tecnologia: resultado.tecnologia,
        dataDeInicioProcesso: resultado.dataDeInicioProcesso,
        dataDeInicioCurso: resultado.dataDeInicioCurso,
        dataDeInicioProjeto: resultado.dataDeInicioProjeto,
        dataDeTerminoProcesso: resultado.dataDeTerminoProcesso,
        dataDeTerminoCurso: resultado.dataDeTerminoCurso,
        dataDeTerminoProjeto: resultado.dataDeTerminoProjeto,
      })
    })
    this.mostrarTecnologias()
    this.mostrarProfessor()
    this.mostrarCategorias()
  }

  editarTurma(id: string) {
    const turma: any = {
      nome: this.form.value.nome,
      categoria: this.form.value.categoria,
      professor1: this.form.value.professor1,
      professor2: this.form.value.professor2,
      tecnologia: this.form.value.tecnologia,
      dataDeInicioProcesso: this.form.value.dataDeInicioProcesso,
      dataDeInicioCurso: this.form.value.dataDeInicioCurso,
      dataDeInicioProjeto: this.form.value.dataDeInicioProjeto,
      dataDeTerminoProcesso: this.form.value.dataDeTerminoProcesso,
      dataDeTerminoCurso: this.form.value.dataDeTerminoCurso,
      dataDeTerminoProjeto: this.form.value.dataDeTerminoProjeto,
    };
    this.service.editarTurma(id, turma).then(
      () => {
        this.route.navigate(['/turma/lista'])
        console.log('Turma editada com sucesso');
      },
      (error) => {
        console.log(error);
      }
    )
  }

  mostrarTecnologias() {
    this.tecnologiaService.listarTecnologia().subscribe((doc) => {
      this.tecnologias = []
      doc.forEach((tecnologia:any) => {
        this.tecnologias.push({
          id: tecnologia.payload.doc.id,
          ...tecnologia.payload.doc.data()
        })
      })
    })
  }

  mostrarProfessor() {
    this.professorService.listarProfessor().subscribe((doc) => {
      console.log(doc);
      this.professores = [];
      doc.forEach((element: any) => {
        this.professores.push({
          id: element.payload.doc.id,
          ...element.payload.doc.data(),
        });
      });
    });
  }

  mostrarCategorias() {
    this.categoriaService.listarCategoria().subscribe((doc) => {
      this.categoria = []
      doc.forEach((categoria:any) => {
        this.categoria.push({
          id: categoria.payload.doc.id,
          ...categoria.payload.doc.data()
        })
      })
    })
  }

}
