import { ProfessorService } from './../../servicos/professor.service';
import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { Professor } from 'src/app/modelos/Professor';

@Component({
  selector: 'app-lista-professor',
  templateUrl: './lista-professor.component.html',
  styleUrls: ['./lista-professor.component.css'],
})
export class ListaProfessorComponent implements OnInit {
  professores: Professor[] = [];
  isModal: boolean = false;
  idProfessorParaExcluir!: any;
  filter!: string;

  constructor(
    private professorService: ProfessorService,
    private router: Router
  ) {}

  ngOnInit(): void {
    this.mostrarProfessor();
  }

  mostrarProfessor() {
    this.professorService.listarProfessor().subscribe((doc) => {
      console.log(doc);
      this.professores = [];
      doc.forEach((element: any) => {
        this.professores.push({
          id: element.payload.doc.id,
          ...element.payload.doc.data(),
        });
      });
    });
  }

  excluir() {
    this.professorService.excluirProfessor(this.idProfessorParaExcluir).then(
      () => {
        console.log('Professor excluido');
        this.isModal = false;
      },
      (error) => {
        console.log(error);
      }
    );
  }

  editar(id: any) {
    this.professorService.mostrarProfessorEdit(id);
    this.router.navigate(['/professor/cadastro/' + id]);
  }

  cancelarAcao() {
    this.isModal = false;
  }

  mostrarModal(id: any) {
    this.isModal = true;
    this.idProfessorParaExcluir = id;
  }

  desativar() {
    const professor: any = {
      status: false,
    };
    this.professorService.desativarUmProfessor(this.idProfessorParaExcluir,professor).then(
      () => {
        console.log('Professor desativado com sucesso');
        this.isModal = false;
      },
      (error) => {
        console.log(error);
      }
    );
  }
}
