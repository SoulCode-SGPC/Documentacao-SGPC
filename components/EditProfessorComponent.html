import { Tecnologias } from './../../modelos/Tecnologias';
import { Component, OnInit } from '@angular/core';
import { ProfessorService } from './../../servicos/professor.service';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Router } from '@angular/router';
import { TecnologiasService } from 'src/app/servicos/tecnologias.service';

@Component({
  selector: 'app-edit-professor',
  templateUrl: './edit-professor.component.html',
  styleUrls: ['./edit-professor.component.css'],
})
export class EditProfessorComponent implements OnInit {
  form: FormGroup;
  id!: string;
  tecnologias!: Tecnologias[];
  checked: boolean = true;
  techCheckBox: any = [];

  constructor(
    private service: ProfessorService,
    private fb: FormBuilder,
    private route: Router,
    private activedroute: ActivatedRoute,
    private tecnologiaService: TecnologiasService
  ) {
    this.form = this.fb.group({
      nome: [
        '',
        [
          Validators.required,
          Validators.minLength(3),
          Validators.maxLength(100),
          Validators.pattern(/^(?!À-Ö)[A-Za-z\',\-àáâãçèéêẽíôõóúû ]*$/),
        ],
      ],
      formacao: [
        '',
        [
          Validators.required,
          Validators.pattern(/^(?!À-Ö)[A-Za-z\',\-àáâãçèéêẽíôõóúû ]*$/),
        ],
      ],
      email: ['', [Validators.email, Validators.required]],
      contato: ['', [Validators.required, Validators.pattern(/^\(?[1-9]{2}\)?-? ?(?:[2-8]|9[1-9])[0-9]{3}\-? ?[0-9]{4}$/)]],
      tecnologia: []
    });
  }


  marcados: any = [];
  ngOnInit(): void {
    // let marcados = this.formprof.techCheckBox
    // console.log(marcados);

    this.mostrarTecnologias();

    const id = <any>this.activedroute.snapshot.params['id'];
    this.service.listarUmProfessor(id).subscribe((res: any) => {
      let resultado = { id: res.id, ...res.data() };
      this.id = resultado.id;

      this.marcados = resultado.tecnologia;
      console.log(this.marcados);

      this.form.patchValue({
        nome: resultado.nome,
        formacao: resultado.formacao,
        email: resultado.email,
        contato: resultado.contato,
      });
    });
  }

  editarProfessor(id: string) {
    this.marcados.forEach((e: any) => {
      if (this.techCheckBox.includes(e) == false) {
        this.techCheckBox.push(e);
      }
    });
    const professor: any = {
      nome: this.form.value.nome,
      formacao: this.form.value.formacao,
      email: this.form.value.email,
      contato: this.form.value.contato,
      tecnologia: this.techCheckBox,
    };

    console.log(professor);

    this.service.editarProfessor(id, professor).then(
      () => {
        this.form.reset();
        console.log('Professor editado com sucesso');
        this.route.navigate(['/professor/lista']);
      },
      (error) => {
        console.log(error);
      }
    );
  }

  qualquerNome(event: any) {
    if (this.marcados.includes(event.target.value) == false) {
      if (this.techCheckBox.includes(event.target.value) == false) {
        this.techCheckBox.push(event.target.value);
        console.log(this.techCheckBox);
      } else {
        this.techCheckBox.splice(
          this.techCheckBox.indexOf(event.target.value),
          1
        );
        console.log(this.techCheckBox);
      }
    } else {
      this.marcados.splice(this.marcados.indexOf(event.target.value),1)
    }
  }

  mostrarTecnologias() {
    this.tecnologiaService.listarTecnologia().subscribe((doc) => {
      this.tecnologias = [];
      console.log(this.tecnologias);
      doc.forEach((tecnologia: any) => {
        this.tecnologias.push({
          id: tecnologia.payload.doc.id,
          ...tecnologia.payload.doc.data(),
        });
      });
    });
  }
}
